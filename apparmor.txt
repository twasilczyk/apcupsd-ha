#include <tunables/global>

# ssh root@192.168.1.XYZ -p 22222
# journalctl -f _TRANSPORT="audit" -g 'apparmor="DENIED"'
profile apcupsd flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  signal (send) set=(kill,term,int,hup,cont) peer=*_apcupsd//*,

  # S6-Overlay
  /init rix,
  /bin/busybox ix,
  /run/{,service/} rw,
  /run/{,service/}{,.}s6** krwix,
  /etc/{s6**,fix-attrs.d/,services.d/} r,
  /package/admin/{s6,execline}** rix,

  # Start new profile for service
  /sbin/apcupsd-wrapper.sh cx -> apcupsd_wrapper,
  /sbin/apcupsd cx -> apcupsd_exec,

  profile apcupsd_wrapper flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*apcupsd,

    # S6-Overlay
    /package/admin/{s6,execline}** mrix,
    /run/s6/** rwk,

    # Access required for service functionality
    /sbin/apcupsd-wrapper.sh r,
    /bin/bash rix,
    /bin/busybox ix,
    /bin/echo ix,
    /usr/bin/curl ix,
    /usr/bin/envsubst ix,
    /usr/bin/jq ix,
    /sbin/apcupsd ix,
    /etc/passwd r,
    /dev/tty rw,

    # network access
    /etc/ssl/openssl.cnf r,
    /etc/hosts r,
    /etc/resolv.conf r,
    /etc/nsswitch.conf r,

    # Bashio
    /usr/lib/bashio/** ix,
    /tmp/.bashio** rwk,

    # apcupsd-wrapper.sh envsubst
    /etc/apcupsd/apcupsd.conf rw,

    # syslogd
    /dev/console w,
    /run/syslogd.pid rw,

    # apcupsd
    #/etc/apcupsd/** r,
    /run/apcupsd.pid rw,
  }

  profile apcupsd_exec flags=(attach_disconnected,mediate_deleted) {
  }
}
