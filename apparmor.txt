#include <tunables/global>

# ssh root@192.168.1.XYZ -p 22222
# journalctl -f _TRANSPORT="audit" -g 'apparmor="DENIED"'
profile apcupsd flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  #/bin/** ix,
  #/usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  #/package/** ix,
  #/command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # s6/execline need mmap+read+exec on their commands
  # ADDED START
  /package/**           mrix,
  /command/**           mrix,
  /bin/**               mrix,
  /usr/bin/**           mrix,
  /usr/sbin/**          mrix,
  /sbin/**              mrix,
  # allow sending signals to the inner profile
  signal (send) set=(term,cont,kill,hup,int) peer=f498a6ad_apcupsd//apcupsd_wrapper,
  # (optional: if the inner ever signals the outer)
  signal (receive) set=(term,cont,kill,hup,int) peer=f498a6ad_apcupsd//apcupsd_wrapper,
  #ADDED END

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,

  # Start new profile for service
  /sbin/apcupsd-wrapper.sh cx -> apcupsd_wrapper,

  profile apcupsd_wrapper flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_ADDON_SLUG,

    # Access to options.json and other files within your addon
    /data/** rw,

    # Access to mapped volumes specified in config.json
    /share/** rw,

    # Access required for service functionality
    /sbin/apcupsd-wrapper.sh r,
    /bin/bash rix,
    /bin/echo ix,
    /etc/passwd r,
    /dev/tty rw,

    # ADDED START
    # s6/execline need mmap+read+exec on their commands
    /package/**           mrix,
    /command/**           mrix,
    /bin/**               mrix,
    /usr/bin/**           mrix,
    /usr/sbin/**          mrix,
    /sbin/**              mrix,
    # Allow reading s6 envdir and its files
    /run/s6/container_environment/      rk,
    /run/s6/container_environment/**   r,
    # (optional but harmless: allow other s6 runtime bits as read-only)
    /run/s6/**                         r,
    # Bashio helper (needs read+mmap+exec)
    /usr/lib/bashio/**        mrix,
    # s6/with-contenv sometimes execs helpers directly
    /usr/bin/with-contenv     rix,
    # allow signals from the outer profile
    signal (receive) set=(term,cont,kill,hup,int) peer=f498a6ad_apcupsd,
    # (optional: also allow sending back to the outer profile)
    signal (send)    set=(term,cont,kill,hup,int) peer=f498a6ad_apcupsd,
    # --- curl / TLS / DNS basics ---
    /etc/ssl/openssl.cnf           r,
    /etc/ssl/cert.pem              r,    # Alpine
    /etc/ssl/certs/**              r,    # Debian/Ubuntu bundle dir
    /etc/ssl/ca-bundle.crt         r,    # (often on Alpine as well)
    /etc/ssl/certs/ca-certificates.crt r,# Debian/Ubuntu single bundle
    /etc/hosts                     r,
    /etc/resolv.conf               r,
    /etc/nsswitch.conf             r,
    # Bashio
    /tmp/** rwk,
    # apcupsd configs (template + final)
    /etc/apcupsd/            r,
    /etc/apcupsd/apcupsd.conf          rw,
    # ADDED END
  }
}
