# To debug apparmor denials:
# ssh root@192.168.1.XYZ -p 22222
# journalctl -f _TRANSPORT="audit" -g 'apparmor="DENIED"'

#include <tunables/global>

profile apcupsd flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  signal (send) set=(kill,term,int,hup,cont) peer=*_apcupsd//*,

  # S6-Overlay
  /init rix,
  /bin/busybox ix,
  /run/{,service/} rw,
  /run/{,service/}{,.}s6** krwix,
  /etc/{s6**,fix-attrs.d/,services.d/} r,
  /package/admin/{s6,execline}** rix,

  # Run wrapper within the same context, because double nested contexts are broken on HA Supervisor
  /sbin/apcupsd-wrapper.sh rix,

  # Bashio
  /bin/bash rix,
  /dev/tty rw,
  /etc/passwd r,
  /tmp/.bashio** rwk,
  /usr/lib/bashio/** ix,

  profile old_wrapper flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*apcupsd,

    # S6-Overlay
    /package/admin/{s6,execline}** mrix,
    /run/s6/** rwk,

    /bin/busybox ix,
    /bin/echo ix,
    /usr/bin/curl ix,
    /usr/bin/envsubst ix,
    /usr/bin/jq ix,
    
    # network access
    /etc/ssl/openssl.cnf r,
    /etc/hosts r,
    /etc/resolv.conf r,
    /etc/nsswitch.conf r,

    # apcupsd-wrapper.sh envsubst
    /etc/apcupsd/apcupsd.conf.in r,
    /etc/apcupsd/apcupsd.conf w,

    # syslogd
    /dev/console w,
    /run/syslogd.pid rw,

    # apcupsd
    #/etc/apcupsd/** r,
    #/run/apcupsd.pid rw,

    /sbin/apcupsd r,
    /sbin/apcupsd px,
  }

  profile newapc flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    /run/apcupsd.pid rw,
  }
}
